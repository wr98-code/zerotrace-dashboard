<script>
// Inisialisasi Peta
var map = L.map('map').setView([-2.5, 118], 5);
L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png', {
  attribution: '&copy; OpenStreetMap contributors'
}).addTo(map);
var markers = [];

// Fungsi untuk menambah titik di peta
function addPoint(lat, lon, info) {
  var marker = L.marker([lat, lon]).addTo(map);
  marker.bindPopup(info).openPopup();
  markers.push(marker);
  map.setView([lat, lon], 13);
  
  // Update live events
  var eventsDiv = document.getElementById('live-events');
  var now = new Date().toLocaleTimeString('id-ID');
  eventsDiv.innerHTML = '<p class="log-entry"><span class="log-time">[' + now + ']</span> Target terdeteksi di ' + info + ' (Lat: ' + lat + ', Lon: ' + lon + ')</p>' + eventsDiv.innerHTML;
  
  // Update connection log
  var logDiv = document.getElementById('connection-log');
  logDiv.innerHTML = '<p class="log-entry"><span class="log-time">[' + now + ']</span> Target terhubung! Lokasi: ' + info + '</p>' + logDiv.innerHTML;
  
  // Update device info
  document.getElementById('device-info').innerHTML = 'IP: ' + (info.match(/IP: ([\d.]+)/) ? info.match(/IP: ([\d.]+)/)[1] : 'N/A') + '<br>Koord: ' + lat + ', ' + lon;
}

// Simulasi: begini nanti kalau target klik JM Store
setTimeout(function() {
  // Ganti koordinat sesuai lokasi target nanti
  addPoint(-7.25, 112.75, "Surabaya - IP: 182.253.x.x - Provider: Indosat");
}, 3000);

console.log('[ZeroTrace] Dashboard siap. Menunggu koneksi dari JM Store...');
